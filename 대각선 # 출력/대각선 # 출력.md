# 문제
주어진 텍스트 그대로 출력
#++++
+#+++
++#++
+++#+
++++#
+++#+
++#++
+#+++
#++++



```c
	int i, j = 0;

	for (i = 0; i < 5; i++) {
		if (i != 0) {
			for (j = 0; j < i; j++) {
				printf("+");
			}
		}
		printf("#");
		for (j = i + 1; j < 5; j++) {
			printf("+");
		}
		printf("\n");
	}

	for (i = 3; i >= 0; i--) {
		if (i != 0) {
			for (j = 0; j < i; j++) {
				printf("+");
			}
		}
		printf("#");
		for (j = i + 1; j < 5; j++) {
			printf("+");
		}
		printf("\n");
	}

```
